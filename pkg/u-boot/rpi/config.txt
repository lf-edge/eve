# Get more options/information on http://elinux.org/RPiconfig
# or on https://www.raspberrypi.org/documentation/configuration/config-txt.md

# Our kernels are located on a Linux partition. Chainload U-Boot to load them.
kernel=u-boot.bin

# Turbo mode: 0 = enable dynamic freq/voltage - 1 = always max
force_turbo=0
# Start in turbo mode for 30 seconds or until cpufreq sets a frequency
initial_turbo=30

# DO NOT overvoltage manually to not void warranty!
over_voltage=0

# Fix mini UART input frequency, and setup/enable up the UART.
enable_uart=1

[pi4]
# Uncomment dtoverlay=raspberrypi-uno-220 if you use advantech uno-220 board,
# this overlay overwrites the SMBIOS information.
#dtoverlay=raspberrypi-uno-220

# SLB9670 TPM over SPI. The overlay differs per board, so gate with filters.
#
# Pi 4 (BCM2711): EVE's bit-banged (spi-gpio) overlay, which u-boot can drive
# for measured boot. Some boards use a different ce(address): default ce=0x01;
# Advantech UNO-220 needs ce=0x00.
# WARNING: Comment this line for OnLogic FR201 device.
dtoverlay=bcm2711-spi-tpm-slb9670,ce=0x01

# Pi 5 (BCM2712): the bit-banged Pi 4 overlay can't bind here since the GPIO/SPI
# controllers live on the RP1 chip (over PCIe), not in the SoC. So use the
# stock, board-agnostic tpm-slb9670 overlay; overlay_map.dtb binds it to RP1
# hardware SPI0 at CE1 (GPIO7) on the same header pins, giving the Linux kernel
# the TPM. There is no support for the RP1 chip on u-boot yet, so the TPM won't
# be available for boot measurement.
[pi5]
dtoverlay=tpm-slb9670

# Uncomment to enable the PCIe x1 port (used by NVMe HATs)
#dtparam=pciex1

[all]

# KMS DRM driver
dtoverlay=vc4-kms-v3d
max_framebuffers=2
hdmi_group=2 # 1=CEA (TVs), 2=DMT (monitors)
hdmi_mode=82 # 82 = 1080p60 in DMT
hdmi_force_hotplug=1
framebuffer_depth=32
framebuffer_ignore_alpha=1

# Disable warning overlays as they don't work well together with linux's graphical output
avoid_warnings=1

# overscan is only needed on old TV sets and if analog gear is in the chain (e.g. DPI)
disable_overscan=1

# Boot in AArch64 mode
arm_64bit=1

# Increase default turbo-mode clock to 1.8GHz
arm_boost=1

# Allow user overrides of config options
include extraconfig.txt

# Uncomment for the following line for OnLogic FR201
# Please, don't forget to disable overlay bcm2711-spi-tpm-slb9670
#include fr201.txt
